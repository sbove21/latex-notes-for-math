	\section{The Basic Simplification Techniques}
	
	\centering
	\textbf{The product rule:}
	
	If \textit{m} and \textit{n} are integers and \textit{a} is a real number, then:
	\begin{equation}
		a^m * a^n = a^m+n
	\end{equation}
